// --------------------FLAPPY BIRD STRUCTURE-----------------------

Backend:

void init(coord_t* pcol, coord_t* pbird);
// Initializes all the variables of the game at the beginning of the program.

void col_mov(coord_t* pcol);
// Changes the x-coordinate of each column per frame and resets the hole position each time the column reappears on the screen.

void Bird_mov(coord_t* pbird);
// Changes the bird's position depending on whether a key was pressed, and if so, which key was pressed.

char Colision(coord_t* pcol, coord_t* pbird);
// Returns 1 if the bird's position will collide with the columns in the next frame; otherwise, returns 0.

void data_track(menue_t* pmen);
// Updates the data that needs to be collected during the game, such as lives and score.

int main_menue(void);
// Returns a different code depending on what the player selects. This is for the menu that appears at the beginning and when the player loses.

int pause_menue(void);
// Returns a different code depending on what the player selects. This is for the menu that appears when the player presses the pause key.

Frontend:

void display_col(coord_t* pcol);
// Renders the columns on the screen based on their coordinates and hole positions.

void display_bird(coord_t* pbird);
// Renders the bird on the screen using its current coordinates.

void display_main_men(void);
// Displays the main menu with the available options when the game starts or ends.

void display_pause_men(void);
// Displays the pause menu with the available options when the game is paused.
